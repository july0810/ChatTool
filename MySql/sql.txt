use test;create table user(uid varchar(8) not null,password varchar(20) not null,nickname varchar(40) not null,primary key(uid));create table message(mes_id varchar(5) not null,mes_text varchar(200) not null,primary key(mes_id));create table send(send_uid varchar(8) not null,sdate date not null,stime time not null,mes_id varchar(5) not null,primary key(send_uid, sdate, stime, mes_id),foreign key(send_uid) references user(uid),foreign key(mes_id) references message(mes_id));create table receive(receive_uid varchar(8) not null,rdate date not null,rtime time not null,mes_id varchar(5) not null,flag char(1) not null,primary key(receive_uid, rdate, rtime, mes_id),foreign key(receive_uid) references user(uid),foreign key(mes_id) references message(mes_id));
